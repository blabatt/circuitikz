\subsection*{Coordinates}

\code{\textbackslash coordinate(mycoord) at(1,1);}\\
\code{\textbackslash draw(mycoord)-{}-(2,2) coordinate(mycoord2);}\\[-1mm]

\textit{Create new \textbf{coordinate system}:}\\
\code{\textbackslash begin\{scope\}[xshift=6cm]} ... \\

\textit{$\exists$ various ways to reference coordinates:} \\
\begin{tabularx}{4cm}{l l}
    \underline{type} & \underline{example} \\
    named & (my\_coord) \\
    anchor & (my\_node.north east) \\
    absolute & (1,-3) \\
    relative & ++(2,2) \\
    tween & (node\_a)!.25!(node\_b) \\
    math & \$(my\_name) + (1,3)\$ \\
    intersect & (<coor\_a> -| <coord\_b>)
\end{tabularx} \\

\entry{43mm}{\textbackslash draw[](A) -{}- ( \$(A)!.85!(B)\$ );}{tweening} \\[-2mm]
