%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection*{Multiplexers}
\textit{Customize the number of pins left (}\texttt{NL}\textit{), bottom (}\texttt{NB}\textit{) and right(}\texttt{NR}\textit{); and relative size left (}\texttt{Lh}\textit{) and right (}\texttt{Rh}\textit{):}\\
\code{demux/.style=\{muxdemux muxdemux }\\
\code{def=\{Lh=4, Rh=8, NL=1, NB=3, NR=8\}\}}\\
\textit{Can create insets with } \texttt{w}, \texttt{inset w}, \texttt{inset Lh}:\\
\code{one bit adder/.style=\{muxdemux muxdemux }\\
\code{\footnotesize def=\{... w=1.5, inset w=.5, inset Lh=2, inset Rh=1.5\}\}}\\
\resizebox{3cm}{!}{\input{circuits/mux}}
\vspace{2mm}

    \item num pins
    \item hide numbers
    \item thickness
\end{itemize}\end{multicols}
}

\subsection*{Seven-Segment Displays}
\textit{Options $\in$:}
{\footnotesize \begin{multicols}{3}\begin{itemize}[label={}]
    \item width
    \item thickness
    \item segment sep
    \item box sep
    \item color on
    \item color off
\end{itemize}\end{multicols}
}
\code{\textbackslash ctikzset\{sevenseg/color on=black\}}\\

\textit{Examples:}\\
\code{\textbackslash foreach \textbackslash i in \{0,...,15\} \textbackslash path (\textbackslash i,0)}\\
\code{\phantom{xx}node[seven segment val=\textbackslash i dot on box off]\{\};}\\
\code{\textbackslash foreach \textbackslash i in \{0,...,15\} \textbackslash path(\textbackslash i/2,-1)}\\
\code{\phantom{xx}node[seven segment val=\textbackslash i dot off box off,}\\
\code{\phantom{xx}fill=gray!30!white]\{\};}\\
