\subsection*{Mirroring, Inversion, Rotation}
\textit{Can } \texttt{mirror} \textit{ or } \texttt{invert} \textit{ circuit elements drawn into paths, and } \texttt{rotate} \textit{ any node elements:}
\entry{45mm}{\textbackslash draw (0,0) to[pD, mirror] (2,0);}{mirroring}\\
\entry{45mm}{\textbackslash draw (0,0) to[pD, invert] (2,0);}{inverting}\\
\entry{45mm}{\textbackslash node[not port, rotate=-90] \{\};}{rotating}\\[1mm]

\textit{Use } \texttt{ctikzflip-} \textit{ macro for \say{unflipping} text:}\\
\code{\textbackslash node[op amp, xscale=-1] \{\textbackslash ctikzflipx\{OA2\} \};}\\
